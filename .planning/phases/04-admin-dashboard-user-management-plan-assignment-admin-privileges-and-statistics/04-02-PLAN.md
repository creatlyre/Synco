  **1. Create `app/admin/views.py`:**
  Follow the existing pattern from `app/dashboard/routes.py` and `app/billing/views.py`.

  ```python
  router = APIRouter(prefix="/admin", tags=["admin-views"])

  @router.get("", response_class=HTMLResponse)
  async def admin_dashboard(request, user=Depends(get_admin_user), db=Depends(get_db)):
      service = AdminService(db)
      stats = service.get_stats()
      context = {"request": request, "user": user, "stats": stats, ...}
      inject_template_i18n(context, request)
      set_locale_cookie_if_param(request, response_context=context)
      return templates.TemplateResponse("admin_dashboard.html", context)

  @router.get("/users", response_class=HTMLResponse)
  async def admin_users_page(request, offset=0, limit=50, search=None, user=Depends(get_admin_user), db=Depends(get_db)):
      service = AdminService(db)
      users = service.list_users(offset, limit, search)
      total = service.admin_repo.count_users()
      context = {"request": request, "user": user, "users": users, "total": total, "offset": offset, "limit": limit, "search": search or "", ...}
      inject_template_i18n(context, request)
      return templates.TemplateResponse("admin_users.html", context)

  @router.get("/users/{user_id}", response_class=HTMLResponse)
  async def admin_user_detail_page(request, user_id: str, user=Depends(get_admin_user), db=Depends(get_db)):
      service = AdminService(db)
      detail = service.get_user_detail(user_id)
      if not detail: raise HTTPException(404)
      context = {"request": request, "user": user, "target_user": detail, ...}
      inject_template_i18n(context, request)
      return templates.TemplateResponse("admin_user_detail.html", context)
  ```

  For non-admin access, `get_admin_user` will raise 403 — add an exception handler in the views file or main.py that redirects HTML requests to /dashboard when 403 is raised on admin routes.
